            topic="raw_telemetry",
            value=request.json()
        )

        # 3. Consultar modelo
        features = await feature_store.latest(request.vin)
        prediction = ml_model.predict(features)

        # 4. Generar gu√≠a AR
        ar_guide = ARGenerator.for_code(
            prediction.fault_code,
            lang=request.language
        )

        return DiagnosisResponse(
            prediction=prediction,
            ar_guide=ar_guide
        )
